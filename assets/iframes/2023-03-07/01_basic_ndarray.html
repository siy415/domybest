import numpy as np

data = np.random.randn(2, 3)
data

array([[ 0.47898231, -0.63179148, -2.11540435],
       [ 0.78996616, -1.14994409, -0.44613141]])
# ndarray 전체에 10을 곱한 것
data * 10

array([[  4.78982313,  -6.31791481, -21.15404353],
       [  7.89966165, -11.49944088,  -4.46131412]])
# ndarray + ndarray, ndarray 행렬의 같은 위치끼리 덧셈이 가능하다.
data + data

array([[ 0.95796463, -1.26358296, -4.23080871],
       [ 1.57993233, -2.29988818, -0.89226282]])
# .shape로 각 차원의 크기를 알 수 있다.
data.shape

(2, 3)
# ndarray는 다양한 자료형을 저장할 수 있으며 .dtype객체로 알 수 있다.
data.dtype

dtype('float64')
