import numpy as np

# np.<kind>, np.<kind>_like: 주어진 모양과 dtype을 갖는 배열 생성 후 1로 초기화, <kind>_like의 경우 입력을 ndarray로 받음
# kind = ["ones", "zeros", "empty", "full"]

# 1. ones
np.ones((2,2))
np.ones_like()

array([[1., 1.],
       [1., 1.]])
# 2. zeros
np.zeros((2,2))

array([[0., 0.],
       [0., 0.]])
# 3. empty
# 쓰레기 값이 들어가있으므로 사용에 주의
np.empty((2,2))

array([[0., 0.],
       [0., 0.]])
# 4. full
np.full((2,2), 99)

array([[99, 99],
       [99, 99]])
# np.eye, np.identity: N x M 단위행렬을 생성, M의 default 값은 N이나, 값을 지정할경우 M-N개의 열이 생성됨
np.eye(2)

array([[1., 0., 0.],
       [0., 1., 0.]])
